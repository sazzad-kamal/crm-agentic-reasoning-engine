## DATABASE SCHEMA

```sql
CREATE TABLE companies (
    company_id INT PRIMARY KEY,
    name VARCHAR,
    status VARCHAR CHECK (status IN ('Active', 'Churned', 'Trial')),
    plan VARCHAR,
    account_owner VARCHAR,
    industry VARCHAR,
    segment VARCHAR CHECK (segment IN ('Enterprise', 'Mid-Market', 'SMB')),
    region VARCHAR,
    renewal_date DATE,
    health_flags VARCHAR CHECK (health_flags IN ('healthy', 'at-risk-low-activity', 'at-risk-renewal-soon'))
);

CREATE TABLE contacts (
    contact_id INT PRIMARY KEY,
    company_id INT REFERENCES companies(company_id),
    first_name VARCHAR,
    last_name VARCHAR,
    email VARCHAR,
    phone VARCHAR,
    job_title VARCHAR,
    role VARCHAR CHECK (role IN ('Decision Maker', 'Champion', 'Technical Contact', 'Executive')),
    lifecycle_stage VARCHAR CHECK (lifecycle_stage IN ('Lead', 'MQL', 'SQL', 'Customer'))
);

CREATE TABLE opportunities (
    opportunity_id INT PRIMARY KEY,
    company_id INT REFERENCES companies(company_id),
    primary_contact_id INT REFERENCES contacts(contact_id),
    name VARCHAR,
    stage VARCHAR CHECK (stage IN ('New', 'Discovery', 'Qualified', 'Proposal', 'Negotiation', 'Closed Won', 'Closed Lost')),
    type VARCHAR CHECK (type IN ('New Business', 'Renewal', 'Expansion')),
    value DECIMAL,
    owner VARCHAR,
    expected_close_date DATE,
    days_in_stage INT
);

CREATE TABLE activities (
    activity_id INT PRIMARY KEY,
    company_id INT REFERENCES companies(company_id),
    contact_id INT REFERENCES contacts(contact_id),
    opportunity_id INT REFERENCES opportunities(opportunity_id),
    type VARCHAR CHECK (type IN ('Call', 'Email', 'Meeting', 'Task')),
    subject VARCHAR,
    due_datetime DATETIME,
    owner VARCHAR,
    priority VARCHAR CHECK (priority IN ('High', 'Medium', 'Low')),
    status VARCHAR CHECK (status IN ('Pending', 'Completed', 'Cancelled'))
);

CREATE TABLE history (
    history_id INT PRIMARY KEY,
    company_id INT REFERENCES companies(company_id),
    contact_id INT REFERENCES contacts(contact_id),
    opportunity_id INT REFERENCES opportunities(opportunity_id),
    type VARCHAR CHECK (type IN ('Call', 'Email', 'Meeting', 'Note')),
    subject VARCHAR,
    source VARCHAR,
    occurred_at DATETIME,
    owner VARCHAR
);
```
