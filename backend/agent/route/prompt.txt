## SCHEMA
companies(company_id, name, status, plan, account_owner, industry, segment, region, renewal_date, health_flags)
  - health_flags: 'healthy' | 'at-risk-low-activity' | 'at-risk-renewal-soon'
  - status: 'Active' | 'Churned' | 'Trial'

contacts(contact_id, company_id, first_name, last_name, email, phone, job_title, role, lifecycle_stage)
  - role: 'Decision Maker' | 'Champion' | 'Technical Contact' | 'Executive'

opportunities(opportunity_id, company_id, primary_contact_id, name, stage, type, value, owner, expected_close_date, days_in_stage)
  - stage: 'New' | 'Discovery' | 'Qualified' | 'Proposal' | 'Negotiation' | 'Closed Won' | 'Closed Lost'
  - type: 'New Business' | 'Renewal' | 'Expansion'

activities(activity_id, company_id, contact_id, opportunity_id, type, subject, due_datetime, owner, priority, status)
history(history_id, company_id, contact_id, opportunity_id, type, subject, source, occurred_at, owner)

## KEYS
- contacts.company_id → companies.company_id
- opportunities.company_id → companies.company_id
- opportunities.primary_contact_id → contacts.contact_id
- activities.company_id → companies.company_id
- activities.contact_id → contacts.contact_id
- activities.opportunity_id → opportunities.opportunity_id
- history.company_id → companies.company_id
- history.contact_id → contacts.contact_id
- history.opportunity_id → opportunities.opportunity_id

## EXAMPLE
Question: "How's my pipeline by stage?"
SQL:
SELECT o.stage, c.name AS company, o.value, c.health_flags
FROM opportunities o
JOIN companies c ON o.company_id = c.company_id
WHERE o.owner = 'jsmith' AND o.stage NOT IN ('Closed Won', 'Closed Lost')
ORDER BY o.value DESC
LIMIT 50
